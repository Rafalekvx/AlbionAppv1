/****** Script for SelectTopNRows command from SSMS  ******/
SELECT TOP (1000) start_run.id
      ,start_run.current_silver
	  ,end_run.new_silver
      ,start_run.current_item_value
	  ,end_run.new_item_value
	  ,start_run.start_time
      ,[end_time]
	  ,(end_run.new_silver - start_run.current_silver) + (end_run.new_item_value-start_run.current_item_value) AS profit
	  ,DATEDIFF(minute,start_time,end_time) as timer
  FROM [AlbionApp].[dbo].[end_run] inner join AlbionApp.dbo.start_run on  end_run.start_id=start_run.id where start_id = 1
